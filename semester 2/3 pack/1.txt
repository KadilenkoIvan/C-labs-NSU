int priority_check(char letter)
{
    if (letter == '(')
    {
        return 1;
    }

    if (letter == ')')
    {
        return 2;
    }

    if (letter == '=')
    {
        return 3;
    }

    if (letter == '+' || letter == '-')
    {
        return 4;
    }

    if (letter == '*' || letter == '/')
    {
        return 5;
    }

    return 0;
}

}
